                                        # 用户模块

                                        # 1-用户登录

                                        此登录功能只对外提供 RESTful 服务接口，不考虑页面。用户发出一个 HTTP 请求，调用登录接口，并接收两个参数，分别为 userName、password。登录成功，以 JSON 形式返回 user 对象。登录失败则返回失败信息。完成试验后，用 Postman 进行测试。

                                        # IUserMapper.java

                                        在 DAO 层的 IUserMapper.java 文件里编写一个 getUser 方法，从数据库 t_user 表里查询一个 User 类型的对象。

                                        public interface IUserMapper{
    @Select("select * from t_user where userName = #{userName}")
    User getUser(String username);
}

                                        # UserService.java

                                        接下来在 service 层再编写一个登录（login）方法，用于实现登录的业务。这里需要传入两个参数。一个是 userName，一个 password。当用户登录成功时返回 user 对象。

                                        @Service
public class UserService {
    @Autowired
    IUserMapper userMapper;

    /**
     * 登录方法
     * @param username 用户名
     * @param password 用户密码
     * @return  返回登陆者
     */
    public User login(String username,String password) {
        User user = userMapper.getUser(username);
        if (user != null) {
            if (user.getPassword().equals(password)) {
               return  user;
            }
        }
        return null;
    }
}

                                        # UserController.java

                                        最后在 controller 层编写一个对外开放的 Rest 接口。用于接收登录（/login）请求。接收两个参数分别为。 userName、password。

                                        @RestController
public class UserController {
    @Autowired
    UserService userService;

    @PostMapping("/login")
    public Map login(String userName,String password){
        User user = userService.login(userName, password);
        Map<String, Object> map = new HashMap<String, Object>();
        if(user!=null){
            map.put("data",user);
            map.put("state","success");
        }else{
            map.put("data",null);
            map.put("state","error");
            map.put("message","登录失败");
        }

        return map;
    }

}

                                        # 2-用户注册

                                        此注册功能只对外提供 RESTful 服务接口，不考虑页面。用户发出一个 HTTP 请求，调用注册接口，并接收用户信息。注册成功，以 JSON 形式返回执行是否成功的相关信息 。完成试验后，用 Postman 进行测试。

                                        # IUserMapper.java

                                        在 DAO 层的 IUserMapper.java 文件里编写一个 saveUser 方法，向数据库 t_user 表插入一个用户。

                                        public interface IUserMapper{
    @Insert("insert into t_user(userName,password,realname,phone) " +
            "values(#{user.userName},#{user.password},#{user.realname},#{user.phone})")
    Integer saveUser(@Param("user") User user);
}

                                        # UserService.java

                                        接下来在 service 层再编写一个 save 方法，用于实现添加用户功能。传入的参数是用户信息。

                                        @Service
public class UserService {
    @Autowired
    IUserMapper userMapper;

    public Boolean save(String userName, String password,
        String realname, String phone) {
        User user = new User();
        user.setUserName(userName);
        user.setPassword(password);
        user.setPhone(phone);
        user.setRealname(realname);
        return userMapper.saveUser(user) > 0;
    }
}

                                        # UserController.java

                                        最后在 controller 层编写一个对外开放的 Rest 接口。用于接收注册（/regist）请求。

                                        @RestController
public class UserController {
    @Autowired
    UserService userService;

    @PostMapping("/regist")
    public Map<String, Object> regist(String userName,String password,
                                      String realname,String phone){
        Map<String, Object> map = new HashMap<String, Object>();
        Boolean b =  userService.save(userName,password,realname,phone);
        if(b){
            map.put("state","success");
        }else {
            map.put("state","error");
            map.put("message","注册失败");
        }
        return map;
    }
}

                                        public interface IUserMapper{
    @Select("select * from t_user where userId = #{userId}")

public class UserService {
