                                            # 优化

                                            # 1-登录验证

                                            用 Spring MVC 的拦截器来实现登录验证功能。如果用户没有登录，则不能访问受保护的资源。比如用户在没有登录的状态下，无法修改个人信息、无法添加购物车、无法生成订单等。

                                            要想验证用户是否登录，写个拦截器就能搞定了。但是登录的信息怎么保存到服务端呢。思考一下，如果用户登录成功，写一个算法，把用户、登录时间、登录过期时间等信息通过加密生成一个唯一的令牌。把令牌响应给客户端并保存。用户访问受保护的资源时，只需要携带令牌即可。后端拿到令牌再解密，解密出来的信息有，用户信息、登录时间、登录过期时间。这样我就能知道此令牌是否可用了。如果令牌不对，查不出来用户数据、如果令牌过期也视为不可用。这样的加密方法我们也不需要自己编写，导入一个 JWT 的 JAR 包，直接调用里面的 API 就能够实现。

                                            # pom 引入 JWT

                                            我们需要 JWT 和 Spring MVC 的拦截器。所以我们在编码之前，先把 JWT 以及 Servlet 相关的依赖导入进来。

                                            导入的方式直接把下面的代码加入到 pom.xml 文件里面 dependencies 标签里即可。

                                                 <!--jwt令牌依赖-->
    <dependency>
      <groupId>io.jsonwebtoken</groupId>
      <artifactId>jjwt</artifactId>
      <version>0.9.1</version>
    </dependency>
    <!--javax xml 相关的依赖 jwt 里有用到-->
    <dependency>
      <groupId>javax.xml.bind</groupId>
      <artifactId>jaxb-api</artifactId>
      <version>2.2.2</version>
    </dependency>

    <!-- Servlet API 包依赖 -->
    <dependency>
      <groupId>javax.servlet</groupId>
      <artifactId>javax.servlet-api</artifactId>
      <version>3.1.0</version>
      <scope>provided</scope>
    </dependency>

                                            # 创建 JwtUtils.java 命令

                                            接着，我们需要添加一个 JwtUtils 类，里面调用 JWT 相关 API，实现生成令牌（token）和获取令牌的功能。下面为 JWT 相关代码，里面有两个重要的方法，生成令牌（generateJsonWebToken）方法、获取令牌（checkJWT）方法。

touch src/main/java/com/utils/JwtUtils.java

                                            # JwtUtils.java 类参考代码

                                            package com.utils;
import io.jsonwebtoken.Claims;
import io.jsonwebtoken.Jwts;
import io.jsonwebtoken.SignatureAlgorithm;
import java.util.Date;
public class JwtUtils {
    //设置token的过期时间
    public static final long EXPIRITION = 1000 * 24 * 60 * 60 * 7;
    //秘钥
    public static final String APPSECRET_KEY = "secret";
    public static String generateJsonWebToken
        (Long userId,String userName,String userImgUrl){
        String token = Jwts.builder().setSubject("SUBJECT")
                .claim("userId", userId)
                .claim("userName",userName)
                .claim("userImgUrl", userImgUrl)
                .setIssuedAt(new Date())
                .setExpiration(new Date(System.currentTimeMillis() + EXPIRITION ))
                .signWith(SignatureAlgorithm.HS256, APPSECRET_KEY).compact();
        return token;
    }

    /**
     * 解密token获取用户信息
     * @param token
     * @return 返回  claims
     */
    public static Claims checkJWT(String token){
        try {
            final Claims claims = Jwts.parser().setSigningKey(APPSECRET_KEY)
                .parseClaimsJws(token).getBody();
            return claims;
        } catch (Exception e) {
            e.printStackTrace();
            return null;
        }
    }
}

                                            # 创建拦截器类 LoginInterceptor.java

                                            再编写一个拦截器类，用于拦截用户、验证用户是否登录功能。在拦截器里获取令牌（token）。然后校验 token 是否有效。下面我们来看一下具体的实现代码。

touch src/main/java/com/interceptor/LoginInterceptor.java

                                            # LoginInterceptor.java 代码参考

                                            public class LoginInterceptor implements HandlerInterceptor {
    /**
     * 模拟拦截用户请求的url，并从url中读取前端传来的token，如果校验通过，则允许进行后续其他的操作
     */
    public boolean preHandle(HttpServletRequest request, HttpServletResponse response,
                             Object handler)
            throws Exception {
        String token = request.getHeader("authorization");
        Claims claims = JwtUtils.checkJWT(token);
        if (claims == null) {
            PrintWriter printWriter = response.getWriter();
            printWriter.println("403");
            printWriter.flush();
            printWriter.close();
        }
        return claims != null;
    }

    public void postHandle(HttpServletRequest httpServletRequest,
                           HttpServletResponse httpServletResponse,
                           Object o, ModelAndView modelAndView) throws Exception {

    }

    public void afterCompletion(HttpServletRequest httpServletRequest,
                                HttpServletResponse httpServletResponse,
                                Object o, Exception e) throws Exception {

    }
}

                                            # 在 springmvc.xml 文件里添加如下配置

                                            创建完拦截器类，我们需要配置一下才能生效。

                                            <!-- 配置拦截器 -->
<mvc:interceptors>
    <mvc:interceptor>
        <!-- 配置拦截器作用的路径 -->
        <mvc:mapping path="/**" />
        <!-- 配置不需要拦截作用的路径 -->
        <mvc:exclude-mapping path="/login" />
        <mvc:exclude-mapping path="/regist"/>
        <mvc:exclude-mapping path="/show*"/>
        <!-- 定义<mvc:interceptor>元素中，表示匹配指定路径的请求才进行拦截 -->
        <bean class="com.interceptor.LoginInterceptor" />
    </mvc:interceptor>
</mvc:interceptors>

                                            # UserService.java

                                            下面为 UserService.java 文件也是实验【用户登录】中的 UserService 类，没有做任何改变。如果是基于保存环境，这个文件无需做任何修改。

                                            @Service
public class UserService {
    @Autowired
    IUserMapper userMapper;
    /**
     * 登录方法
     * @param username 用户名
     * @param password 用户密码
     * @return  返回登陆者
     */
    public User login(String username,String password) {
        User user = userMapper.getUser(username);
        if (user != null) {
            if (user.getPassword().equals(password)) {
               return  user;
            }
        }
        return null;
    }
}

                                            # UserController.java

                                            最后，我们在 UserController.java 的登录方法中修改一下代码。把登录成功的用户信息，生成令牌（token）并响应给客户端。

                                            @RestController
public class UserController {
    @Autowired
    UserService userService;

    @PostMapping("/login")
    public Map login(String userName,String password){
        User user = userService.login(userName, password);
        Map<String, Object> map = new HashMap<String, Object>();
        if(user!=null){
            map.put("data",user);
            map.put("state","success");
            //保存 token
            String token =  JwtUtils.generateJsonWebToken
                    (user.getUserId(),user.getUserName(),user.getUserImgUrl());
            //把 token 放到 map 对象里，响应给客户端
            map.put("token",token);
        }else{
            map.put("data",null);
            map.put("state","error");
            map.put("message","登录失败");
        }

        return map;
    }
}
